141 public function parse(): void
142 {
143 $this->lineNumber = 1;
144 $this->charNumber = 1;
145 $eof = false;
146
147 while (!feof($this->stream) && !$eof) {
148 $pos = ftell($this->stream);
149 // set the underlying streams chunk size, so it delivers according to the request from stream_get_line
150 stream_set_chunk_size($this->stream, $this->bufferSize);
151 $line = stream_get_line($this->stream, $this->bufferSize, $this->lineEnding);
152
153 if (false === $line) {
154 $line = '';
155 }
156
157 $ended = (bool) (ftell($this->stream) - \strlen($line) - $pos);
158 // if we're still at the same place after stream_get_line, we're done
159 $eof = ftell($this->stream) === $pos;
160
161 $byteLen = \strlen($line);
162 for ($i = 0; $i < $byteLen; $i++) {
163 if ($this->listener instanceof PositionAwareInterface) {
164 $this->listener->setFilePosition($this->lineNumber, $this->charNumber);
165 }
166 $this->consumeChar($line[$i]);
167 $this->charNumber++;
168
169 if ($this->stopParsing) {
170 return;
171 }
172 }

175 $this->lineNumber++;

186 private function consumeChar(string $char): void
187 {
188 // see https://en.wikipedia.org/wiki/Byte_order_mark
189 if ($this->charNumber < 5
190 && 1 === $this->lineNumber
191 && $this->checkAndSkipUtfBom($char)
192 ) {
193 return;
194 }
195
196 // valid whitespace characters in JSON (from RFC4627 for JSON) include:
197 // space, horizontal tab, line feed or new line, and carriage return.
198 // thanks: http://stackoverflow.com/questions/16042274/definition-of-whitespace-in-json
199 if ((' ' === $char || "\t" === $char || "\n" === $char || "\r" === $char) &&
200 !(self::STATE_IN_STRING === $this->state ||
201 self::STATE_UNICODE === $this->state ||
202 self::STATE_START_ESCAPE === $this->state ||
203 self::STATE_IN_NUMBER === $this->state)
204 ) {
205 // we wrap this so that we don't make a ton of unnecessary function calls
206 // unless someone really, really cares about whitespace.
207 if ($this->emitWhitespace) {
208 $this->listener->whitespace($char);
209 }
210
211 return;
212 }
213
214 switch ($this->state) {
215 case self::STATE_IN_STRING:
216 if ('"' === $char) {
217 $this->endString();
218 } elseif ('\\' === $char) {
219 $this->state = self::STATE_START_ESCAPE;
220 } elseif ($char < "\x1f") {
221 $this->throwParseError('Unescaped control character encountered: '.$char);
222 } else {
223 $this->buffer .= $char;
224 }
225 break;
226
227 case self::STATE_IN_ARRAY:
228 if (']' === $char) {
229 $this->endArray();
230 } else {
231 $this->startValue($char);
232 }
233 break;
234
235 case self::STATE_IN_OBJECT:
236 if ('}' === $char) {
237 $this->endObject();
238 } elseif ('"' === $char) {
239 $this->startKey();
240 } else {
241 $this->throwParseError('Start of string expected for object key. Instead got: '.$char);
242 }
243 break;
244
245 case self::STATE_END_KEY:
246 if (':' !== $char) {
247 $this->throwParseError("Expected ':' after key.");
248 }
249 $this->state = self::STATE_AFTER_KEY;
250 break;
251
252 case self::STATE_AFTER_KEY:
253 $this->startValue($char);
254 break;
255
256 case self::STATE_START_ESCAPE:
257 $this->processEscapeCharacter($char);
258 break;
259
260 case self::STATE_UNICODE:
261 $this->processUnicodeCharacter($char);
262 break;
263
264 case self::STATE_UNICODE_SURROGATE:
265 $this->unicodeEscapeBuffer .= $char;
266 if (2 === mb_strlen($this->unicodeEscapeBuffer)) {
267 $this->endUnicodeSurrogateInterstitial();
268 }
269 break;
270
271 case self::STATE_AFTER_VALUE:
272 $within = end($this->stack);
273 if (self::STACK_OBJECT === $within) {
274 if ('}' === $char) {
275 $this->endObject();
276 } elseif (',' === $char) {
277 $this->state = self::STATE_IN_OBJECT;
278 } else {
279 $this->throwParseError("Expected ',' or '}' while parsing object. Got: ".$char);
280 }
281 } elseif (self::STACK_ARRAY === $within) {
282 if (']' === $char) {
283 $this->endArray();
284 } elseif (',' === $char) {
285 $this->state = self::STATE_IN_ARRAY;
286 } else {
287 $this->throwParseError("Expected ',' or ']' while parsing array. Got: ".$char);
288 }
289 } else {
290 $this->throwParseError(
291 'Finished a literal, but unclear what state to move to. Last state: '.$within
292 );
293 }
294 break;
295
296 case self::STATE_IN_NUMBER:
297 if (ctype_digit($char)) {
298 $this->buffer .= $char;
299 } elseif ('.' === $char) {
300 if (false !== strpos($this->buffer, '.')) {
301 $this->throwParseError('Cannot have multiple decimal points in a number.');
302 } elseif (false !== stripos($this->buffer, 'e')) {
303 $this->throwParseError('Cannot have a decimal point in an exponent.');
304 }
305 $this->buffer .= $char;
306 } elseif ('e' === $char || 'E' === $char) {
307 if (false !== stripos($this->buffer, 'e')) {
308 $this->throwParseError('Cannot have multiple exponents in a number.');
309 }
310 $this->buffer .= $char;
311 } elseif ('+' === $char || '-' === $char) {
312 $last = mb_substr($this->buffer, -1);
313 if (!('e' === $last || 'E' === $last)) {
314 $this->throwParseError("Can only have '+' or '-' after the 'e' or 'E' in a number.");
315 }
316 $this->buffer .= $char;
317 } else {
318 $this->endNumber();
319 // we have consumed one beyond the end of the number
320 $this->consumeChar($char);
321 }
322 break;
323
324 case self::STATE_IN_TRUE:
325 $this->buffer .= $char;
326 if (4 === \strlen($this->buffer)) {
327 $this->endTrue();
328 }
329 break;
330
331 case self::STATE_IN_FALSE:
332 $this->buffer .= $char;
333 if (5 === \strlen($this->buffer)) {
334 $this->endFalse();
335 }
336 break;
337
338 case self::STATE_IN_NULL:
339 $this->buffer .= $char;
340 if (4 === \strlen($this->buffer)) {
341 $this->endNull();
342 }
343 break;
344
345 case self::STATE_START_DOCUMENT:
346 $this->listener->startDocument();
347 if ('[' === $char) {
348 $this->startArray();
349 } elseif ('{' === $char) {
350 $this->startObject();
351 } else {
352 $this->throwParseError('Document must start with object or array.');
353 }
354 break;
355
356 case self::STATE_END_DOCUMENT:
357 if ('[' !== $char && '{' !== $char) {
358 $this->throwParseError('Expected end of document.');
359 }
360 $this->state = self::STATE_START_DOCUMENT;
361 $this->consumeChar($char);
362 break;
363
364 case self::STATE_DONE:
365 $this->throwParseError('Expected end of document.');
366 break;
367
368 default:
369 $this->throwParseError('Internal error. Reached an unknown state: '.$this->state);
370 break;
371 }

374 private function checkAndSkipUtfBom(string $c): bool
375 {
376 if (1 === $this->charNumber) {
377 if ($c === \chr(239)) {
378 $this->utfBom = self::UTF8_BOM;
379 } elseif ($c === \chr(254) || $c === \chr(255)) {
380 // NOTE: could also be UTF32_BOM
381 // second character will tell
382 $this->utfBom = self::UTF16_BOM;
383 } elseif ($c === \chr(0)) {
384 $this->utfBom = self::UTF32_BOM;
385 }
386 }
387
388 if (self::UTF16_BOM === $this->utfBom && 2 === $this->charNumber &&
389 $c === \chr(254)) {
390 $this->utfBom = self::UTF32_BOM;
391 }
392
393 if (self::UTF8_BOM === $this->utfBom && $this->charNumber < 4) {
394 // UTF-8 BOM starts with chr(239) . chr(187) . chr(191)
395 return true;
396 }
397 if (self::UTF16_BOM === $this->utfBom && $this->charNumber < 3) {
398 return true;
399 }
400 if (self::UTF32_BOM === $this->utfBom && $this->charNumber < 5) {
401 return true;
402 }
403
404 return false;

538 private function processUnicodeCharacter(string $char): void
539 {
540 if (!ParserHelper::isHexCharacter($char)) {
541 $this->throwParseError(
542 'Expected hex character for escaped Unicode character. '
543 .'Unicode parsed: '.implode('', $this->unicodeBuffer).' and got: '.$char
544 );
545 }
546 $this->unicodeBuffer[] = $char;
547 if (4 === \count($this->unicodeBuffer)) {
548 $codepoint = hexdec(implode('', $this->unicodeBuffer));
549
550 if ($codepoint >= 0xD800 && $codepoint < 0xDC00) {
551 $this->unicodeHighSurrogate = $codepoint;
552 $this->unicodeBuffer = [];
553 $this->state = self::STATE_UNICODE_SURROGATE;
554 } elseif ($codepoint >= 0xDC00 && $codepoint <= 0xDFFF) {
555 if (-1 === $this->unicodeHighSurrogate) {
556 $this->throwParseError('Missing high surrogate for Unicode low surrogate.');
557 }
558 $combinedCodepoint = (($this->unicodeHighSurrogate - 0xD800) * 0x400) + ($codepoint - 0xDC00) + 0x10000;
559
560 $this->endUnicodeCharacter($combinedCodepoint);
561 } else {
562 if (-1 !== $this->unicodeHighSurrogate) {
563 $this->throwParseError('Invalid low surrogate following Unicode high surrogate.');
564 } else {
565 $this->endUnicodeCharacter($codepoint);
566 }
567 }
