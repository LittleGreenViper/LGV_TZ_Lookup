124 private static function _wn_PnPoly( $point,
125 $polygon
126 ) {
127 $wn = 0;
128 $n = count($polygon);
129
130 // loop through all edges of the polygon
131 for ($i = 0; $i < $n; $i++) { // edge from polygon[i] to polygon[i+1]
132 if ($polygon[$i][0] <= $point[0]) { // start y <= point[0]
133 if ($polygon[($i + 1) % $n][0] > $point[0]) // an upward crossing
134 if (self::_isLeft($polygon[$i], $polygon[($i + 1) % $n], $point) > 0) // point left of edge
135 ++$wn; // have a valid up intersect
136 } else { // start y > point[0] (no test needed)
137 if ($polygon[($i + 1) % $n][0] <= $point[0]) // a downward crossing
138 if (self::_isLeft($polygon[$i], $polygon[($i + 1) % $n], $point) < 0) // point right of edge
